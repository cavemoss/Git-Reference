Local repository (on your PC)
Remote repository (crated on GitHub)


git remote          # list linked remote repositories
  git remote -v     # give more details

git remote add <remote_repository_name 'origin'> <remote_repository_url>
git remote remove <remote_repository_name>

git config --global credential.username "<username>"


push (upload)
pull (download)

git push <a_linked_remote_repository_name> <commit_branch_name 'master'>              # upload (push)
git push <a_linked_remote_repository_name> <commit_branch_name 'master'> -f           # overwrite an uploaded commit
git <a_linked_remote_repository_name> <commit_branch_name 'master'> --set-upstream    # set a shortcut
  git push

git fetch     # update local repository

git pull <a_linked_remote_repository_name> <commit_branch_name 'master'>
git <a_linked_remote_repository_name> <commit_branch_name 'master'> --set-upstream    # set a shortcut
  git pull
