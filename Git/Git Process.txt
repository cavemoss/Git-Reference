git init    # initialize empty repository

git status                 # list what's new
git log                    # list commit history
  git log --all            # list all the later commits
  git log --all --graph    # visualize branching
 
git add <file>      # pick which files have been added or changed
  git add .         # add all
git reset <file>    # undo add  
  git reset .       # reset all 

git commit                             # commit to the change
  git commit -m "<message>"            # attach a message
  git commit -m "<message>" --amend    # correct previous commit


git config --global user.name "Your Name"             # configure credentians
git config --global user.email "email@example.com"
